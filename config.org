* Keybind

** Tabout
#+begin_src emacs-lisp
(defun my/tabout ()
  (interactive)
  (cond
   ((and (derived-mode-p 'org-mode)
         (fboundp 'org-at-table-p)
         (org-at-table-p))
    (call-interactively #'org-cycle))
   (t
    (ignore-errors (sp-up-sexp)))))
(map! :i "TAB" #'my/tabout)
#+end_src
